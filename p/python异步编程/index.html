import asyncio
from pyppeteer import launch

async def screen_shot_1():
    browser = await launch()
    page = await browser.newPage()
    await page.goto('https://fastapi.tiangolo.com/zh/tutorial/middleware/')
    await page.screenshot({'path': '1.png'})
    await browser.close()

async def screen_shot_2():
    browser = await launch()
    page = await browser.newPage()
    await page.goto('https://www.ruanyifeng.com/blog/2019/11/python-asyncio.html')
    await page.screenshot({'path': '2.png'})
    await browser.close()

async def main():
    await asyncio.gather(screen_shot_1(), screen_shot_2())

asyncio.run(main())
